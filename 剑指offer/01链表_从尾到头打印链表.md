## Code
```cpp
class Solution {
private:
    vector<int> ans;
    void rec(ListNode* head){
        if(head!=nullptr) printListFromTailToHead(head->next);
        else return;
        ans.push_back(head->val);
    }
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        rec(head);
        return ans;
    }
};
```
